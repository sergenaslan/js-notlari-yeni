Node js Çalışma Notları

createServer Server Oluşturma

Aşşağıda En basit haliyle bir node server örneği bulunmaktadır 

const http = require('http');

const server = http.createServer((request, response ) => {
    console.log(request.url);
    response.writeHead(200, { 'Content-Type': 'text/plain; charset=utf-8' });
    response.end('hermaba');
});

server.listen(3000, () => {
    console.log('Server çalışıyor 🚀');
});

/* 
request 
gelen istedğin detaylarını öğrenebileceğimiz fonksiyonların bulunuğu bir nesne
response 
gelen isteğe cevap verirken kullanabileceğimiz fonksiyonların bulunduğu nesne
 */

Router Nesnesi

Router Nesnesi route’ları ayrı dosyalara bölerek projeyi düzenli yapmanı sağlar.

//signIn.js

const express = require('express');
const router = express.Router();

router.get('/signIn', (req, res) => {
    res.send("signIn sayfası");
});

router.post('/signUp', (req, res) => {
    res.send("signUp sayfası");
});

module.exports = router;



 //App.js
const express = require('express');
const app = express();

const signIn = require('./routes/signIn');

app.use('/user', signIn);
// app.use() fonksiyonu, middleware veya router'ları Express uygulamasına eklemek için kullanılır.
app.listen(3000, () => {
    console.log("express server çalıştı.");
});

Middleware Konusu

Middleware, istek (request) ile cevap (response) arasına girip işlem yapan fonksiyondur.
Middleware = İstek geldiğinde arada çalışan yardımcı fonksiyon
  • veri kontrol etmek, log tutmak, yetki kontrolü yapmak, body parse etmek, route’a yönlendirmek

// isLoginMiddleware.js

const isLogin = (req, res, next) => {
    const isLogin = true;

    if (isLogin)
        next();
    else
        res.send("Lütfen giriş yapın.");
};

module.exports = isLogin;


//profile.js

const express = require('express');
const router = express.Router();

// helpers
const isLogin = require('./isLoginMiddleware');

router.get('/', isLogin, (req, res) => {
    res.send("profile sayfası");
});

module.exports = router;  


//App.js
const express = require('express');
const app = express();
const profile = require('./routes/profile');
 
 
app.use('/profile', profile);

 
app.listen(3000, () => {
    console.log("express server çalıştı.");
});
