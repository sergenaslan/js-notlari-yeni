  name: 'Sergen'
})

}

  'Elma',

])

toRef()

toRef(), bir objenin tek bir özelliğini ref haline getirir. Ama yeni bir ref oluşturmaz, orijinal property'ye bağlı (linked) bir ref oluşturur. Bir objenin tek bir değerine reactive bozmadan destructuring yaparak ref ile tanımlanmış gibi kullanabiliriz. 



<script setup>
import { ref, toRef } from 'vue'

const user = ref({
  name: 'Sergen',
  age: '25'
})

function changeName() {
  user.value.name = 'Mehmet'
}

const isim = toRef(user.value, 'name')
const age = toRef(user.value, 'age')


</script>

<template>

  <h1>{{ isim  }} - {{ age  }}< /h1>


  <button @click="changeName">
    Değiştir
  </button>

</template>

toRefs

toRef in yaptğının aynısını çoklu versiyon şeklinde yapar. her birini tek tek yazma durumunda kalmadan yapar. 


<script setup>
import { ref, toRefs } from 'vue'

const user = ref({
  name: 'Sergen',
  age: 25
})

function changeName() {
  user.value.name = 'Aslan'
  user.value.age = 27
}

const { name, age } = toRefs(user.value)

</script>

<template>

  <h1>{{ name  }} - {{ age }}</h1>


  <button @click="changeName">
    Değiştir
  </button>

</template>
